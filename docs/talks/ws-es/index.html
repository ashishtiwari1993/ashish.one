Sample Queries for Elasticsearch Workshop#

CRUD#


# insert

POST meetup/_doc/
{
  "name":"Ashish Tiwari"
}

# insert with id

POST meetup/_doc/1
{
  "name":"Ashish Tiwari"
}

# read

GET meetup/_search

# update

POST meetup/_doc/1
{
  "name":"Ashish",
  "company":"elastic",
  "address":"Navi Mumbai kharghar",
  "skills":{
    "language":["php","java","node"],
    "database":["mysql","mongodb"],
    "search":"elasticsearch"
  }
}

# search with query

GET meetup/_search
{
  "query": {
    "match": {
      "address": "navi"
    }
  }
}

# delete

DELETE meetup
